# AutoQuant-Alpha | Day 5: Margin Logic Alert System

pip install -r requirements.txt
bash scripts/start.sh          # Run margin monitor

bash scripts/verify.sh         # Run tests
bash scripts/cleanup.sh        # Clean artifacts
